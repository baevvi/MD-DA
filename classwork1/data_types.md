1.  Присвоить значение 33.3 переменной my\_numeric.

<!-- -->

    my_numeric <- 33.3
    cat("my_numeric = ", my_numeric)

    ## my_numeric =  33.3

1.  Проверить тип данных переменной.

<!-- -->

    cat(class(my_numeric))

    ## numeric

1.  Сменить значение my\_numeric на 42.

<!-- -->

    my_numeric = 42
    cat("my_numeric = ", my_numeric)

    ## my_numeric =  42

1.  Проверить тип данных my\_numeric.

<!-- -->

    cat(class(my_numeric))

    ## numeric
